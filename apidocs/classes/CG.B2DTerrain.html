CG.B2DTerrain Class

Extends CG.B2DEntity
Defined in: cangaja/box2d/b2d-terrain.js:1
Module: CG

B2DTerrain looks similar to B2DPolygon but has more features for polygon manipulation like clipping and triangulation.


var terrainShapes =
[{
   outer: [{
       x: 0,
       y: 100.5
   }, {
       x: 1024,
       y: 100.5
   }, {
       x: 1024,
       y: 768
   }, {
       x: 0,
       y: 768
   }],
   holes: []
}]

b2world.createTerrain({
    name: 'terrain',
    image: false
    terrainShape: terrainShapes,
    x:0,
    y:0,
    world: b2world,
    scale: 40
})
